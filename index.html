import 'dart:core';

// ---------- Kelas Induk ----------
class Perangkat {
  String kodeInventaris;
  String nama;
  String _status;
  String? _pelaporAktif;

  Perangkat(this.kodeInventaris, this.nama, this._status) {
    _pelaporAktif = null;
  }

  String get status => _status;
  String? get pelaporAktif => _pelaporAktif;

  bool laporkan(String namaPelapor) {
    if (_status == 'Tersedia') {
      _status = 'Rusak';
      _pelaporAktif = namaPelapor;
      return true;
    }
    return false;
  }

  bool selesaikan() {
    if (_status == 'Rusak' || _status == 'Dalam Perbaikan') {
      _status = 'Tersedia';
      _pelaporAktif = null;
      return true;
    }
    return false;
  }

  void tampilkanInfo() {
    print('Kode Inventaris: $kodeInventaris');
    print('Nama Perangkat : $nama');
    print('Status         : $_status');
    if (_pelaporAktif != null) {
      print('Pelapor        : $_pelaporAktif');
    }
  }
}

// ---------- Kelas Turunan ----------
class Komputer extends Perangkat {
  String spesifikasi;
  int ramGB;
  Komputer(String kode, String nama, String status, this.spesifikasi, this.ramGB)
      : super(kode, nama, status);

    print('--- [Info Komputer] ---');
    super.tampilkanInfo();

}

// ---------- Kelas Pelapor ----------
class Pelapor {
  String nama;
  String nik;
  int _jumlahLaporanAktif = 0;
  List<String> _laporanAktif = [];

  Pelapor(this.nama, this.nik);

  int get jumlahLaporanAktif => _jumlahLaporanAktif;
  List<String> get laporanAktif => List.unmodifiable(_laporanAktif);

  Map<String, dynamic> buatLaporan(Perangkat perangkat) {
    if (_jumlahLaporanAktif >= 2) {
      return {'success': false, 'message': 'Batas maksimal 2 laporan aktif per pelapor.'};
    }
    if (perangkat.status != 'Tersedia') {
      return {'success': false, 'message': 'Perangkat ${perangkat.nama} sedang tidak tersedia.'};
    }
    bool berhasil = perangkat.laporkan(this.nama);
    if (berhasil) {
      _jumlahLaporanAktif++;
      _laporanAktif.add(perangkat.kodeInventaris);
      return {'success': true, 'message': 'Laporan berhasil! $nama melaporkan ${perangkat.nama}.'};
    }
    return {'success': false, 'message': 'Gagal melaporkan perangkat.'};
  }

  Map<String, dynamic> selesaikanLaporan(Perangkat perangkat) {
    if (perangkat.status == 'Rusak' || perangkat.status == 'Dalam Perbaikan') {
      int idx = _laporanAktif.indexOf(perangkat.kodeInventaris);
      if (idx != -1) {
        _laporanAktif.removeAt(idx);
        _jumlahLaporanAktif--;
      }
      bool berhasil = perangkat.selesaikan();
      if (berhasil) {
        return {'success': true, 'message': 'Perangkat ${perangkat.nama} telah diperbaiki.'};
      }
    }
    return {'success': false, 'message': 'Perangkat ${perangkat.nama} tidak dalam status rusak.'};
  }

  void tampilkanInfo() {
    print('Nama Pelapor  : $nama');
    print('NIK           : $nik');
    print('Laporan Aktif : $_jumlahLaporanAktif perangkat');
    if (_laporanAktif.isNotEmpty) {
      print('Daftar kode   : ${_laporanAktif.join(', ')}');
    }
  }
}

// ---------- Kelas LaporanKerusakan ----------
class LaporanKerusakan {
  String idLaporan;
  Pelapor pelapor;
  Perangkat perangkat;
  DateTime tanggalLapor;
  DateTime? tanggalSelesai;
  String deskripsi;
  bool isSukses = false;

  LaporanKerusakan(this.idLaporan, this.pelapor, this.perangkat, this.deskripsi)
      : tanggalLapor = DateTime.now();

  void selesaikanLaporan() {
    tanggalSelesai = DateTime.now();
    isSukses = true;
  }

  void tampilkanDetail() {
    String tglLapor = tanggalLapor.toLocal().toString().split(' ')[0];
    String tglSelesai = tanggalSelesai != null
        ? tanggalSelesai!.toLocal().toString().split(' ')[0]
        : 'Belum selesai';
    print('==================================================');
    print('          NOTA LAPORAN KERUSAKAN PERANGKAT         ');
    print('==================================================');
    print('ID Laporan   : $idLaporan');
    print('Pelapor      : ${pelapor.nama} (${pelapor.nik})');
    print('Perangkat    : ${perangkat.nama} [${perangkat.kodeInventaris}]');
    print('Tanggal Lapor: $tglLapor');
    print('Tanggal Selesai: $tglSelesai');
    print('Deskripsi    : ${deskripsi.isNotEmpty ? deskripsi : '-'}');
    print('Status       : ${isSukses ? 'SELESAI ✅' : 'AKTIF ⏳'}');
    print('==================================================');
  }
}

// ALGORITMA SistemPelaporanKerusakan

DEKLARASI
  // Tipe bentukan untuk objek Laporan
  CLASS Laporan
    PRIVATE idLaporan : string
    PRIVATE idPerangkat : string
    PRIVATE deskripsi : string
    PRIVATE tanggal : date
    PRIVATE pelapor : string
    PRIVATE status : string
    PRIVATE waktuDibuat : datetime

    // Konstruktor (Proses OOP: Membuat objek)
    METHOD Constructor(idPerangkat, deskripsi, tanggal, pelapor)
      this.idPerangkat <- idPerangkat
      this.deskripsi <- deskripsi
      this.tanggal <- tanggal
      this.pelapor <- pelapor
      this.idLaporan <- generateUniqueID() // Proses: generate ID otomatis
      this.status <- "Dilaporkan" // Proses: set status awal
      this.waktuDibuat <- currentDateTime() // Proses: catat waktu
    END METHOD

    // Proses: Menyimpan objek ke basis data (simulasi)
    METHOD simpan()
      database.tambahLaporan(this)
    END METHOD

    // Proses: Mengubah status laporan
    METHOD updateStatus(statusBaru)
      this.status <- statusBaru
    END METHOD

    // Fungsi untuk mendapatkan ID laporan (Output)
    METHOD getID() -> string
      RETURN this.idLaporan
    END METHOD
  END CLASS

  // Variabel global untuk simulasi database
  database : array of Laporan
  laporanBaru : Laporan

  // Variabel untuk input
  input_idPerangkat : string
  input_deskripsi : string
  input_tanggal : date
  input_pelapor : string
  dataLengkap : boolean

ALGORITMA UTAMA
  // === MULAI ===
  OUTPUT "=== SISTEM PELAPORAN KERUSAKAN PERANGKAT LAB ==="

  // === INPUT (dengan validasi/perulangan) ===
  REPEAT
    OUTPUT "Masukkan ID Perangkat : "
    INPUT input_idPerangkat

    OUTPUT "Masukkan Deskripsi Kerusakan : "
    INPUT input_deskripsi

    OUTPUT "Masukkan Tanggal Kejadian (YYYY-MM-DD) : "
    INPUT input_tanggal

    OUTPUT "Masukkan Nama Pelapor : "
    INPUT input_pelapor

    // Proses Validasi
    IF (input_idPerangkat == "" OR input_deskripsi == "" OR
        input_tanggal == "" OR input_pelapor == "") THEN
      dataLengkap <- FALSE
      OUTPUT "ERROR: Data tidak lengkap! Silakan isi semua field."
    ELSE
      dataLengkap <- TRUE
    END IF
  UNTIL (dataLengkap == TRUE)

  // === PROSES UTAMA (OOP) ===
  // 1. Membuat objek Laporan dari input
  laporanBaru <- NEW Laporan(input_idPerangkat, input_deskripsi,
                   input_tanggal, input_pelapor)

  // 2. Menyimpan objek ke database
  laporanBaru.simpan()

  // 3. Update status (opsional, karena konstruktor sudah set "Dilaporkan")
  laporanBaru.updateStatus("Dilaporkan")

  // === OUTPUT ===
  OUTPUT "----------------------------------------------"
  OUTPUT "SUKSES! Laporan kerusakan telah dibuat."
  OUTPUT "ID Laporan : " + laporanBaru.getID()
  OUTPUT "Status : Dilaporkan"
  OUTPUT "Waktu Pelaporan : " + laporanBaru.waktuDibuat
  OUTPUT "----------------------------------------------"
  OUTPUT "Terima kasih, laporan akan segera diproses."

  // === SELESAI ===
END ALGORITMA
